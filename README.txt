# To Create a GitHub repository named Project-1-Biocomputing

mkdir "Project-1-Biocomputing" #make folder 
cd "Project-1-Biocomputing/" #change directory
git init #to initializes a local repository in the folder which we just created
ls -a # to check the local repository we just created exists or not


# For 1 subfolder "data"


mkdir "data" #make subfolder 

cd "data" #change directoty

cp /mnt/c/Users/ER.NAZLEEN/OneDrive/Desktop/retrieved_files/*.fasta /home/nazleen/Project-1-Biocomputing/data/

cp /mnt/c/Users/ER.NAZLEEN/OneDrive/Desktop/retrieved_files/*.pdb /home/nazleen/Project-1-Biocomputing/data/  #copyfiles

git status # to see status

git status 

git commit -m "retrieved_files" #records a snapshot of staged files into the repository

git status

git remote add origin git@github.com:NAZLEEN21/Project-1-Biocomputing # to connect local repository to the GitHub

git push -u origin main # to uploads the commits from local repository to GitHub
